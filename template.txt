class NumbersParser < Formula
  include Language::Python::Virtualenv

  desc "Read and write Apple Numbers spreadsheets"
  homepage "https://github.com/masaccio/numbers-parser"
  url "${url}"
  sha256 "${sha256}"

  depends_on "protobuf"
  depends_on "python3"

  def install
    venv = virtualenv_create(libexec, "python3)
    %w[${dependencies}].each do |r|
      venv.pip_install resource(r)
    end
    venv.pip_install_and_link buildpath
  end

  test do
    assert_equal "${version}", shell_output("#{bin}/cat-numbers -V").strip
  end
end
